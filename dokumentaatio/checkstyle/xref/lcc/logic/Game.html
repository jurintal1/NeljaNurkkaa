3   public class Game {

5       private CornerDeck upLeftCorner;
6       private CornerDeck downLeftCorner;
7       private CornerDeck upRightCorner;
8       private CornerDeck downRightCorner;
9   
10      private SideDeck left1;
11      private SideDeck left2;
12      private SideDeck left3;
13      private SideDeck left4;
14      private SideDeck right1;
15      private SideDeck right2;
16      private SideDeck right3;
17      private SideDeck right4;
18  
19      private FoundationDeck clubsUp;
20      private FoundationDeck diamondsUp;
21      private FoundationDeck heartsUp;
22      private FoundationDeck spadesUp;
23      private FoundationDeck clubsDown;
24      private FoundationDeck diamondsDown;
25      private FoundationDeck heartsDown;
26      private FoundationDeck spadesDown;
27      
28      private Deck[] manoeuvreDecks;
29  
30      public Game() {
31          createFoundationDecks();
32          addAcesAndKingsToFoundationDecks();
33          createManoeuvreDecks();
34          dealToManoeuvreDecks();
35      }
36  
37      public final void createManoeuvreDecks() {
38          this.upLeftCorner = new CornerDeck();
39          this.downLeftCorner = new CornerDeck();
40          this.upRightCorner = new CornerDeck();
41          this.downRightCorner = new CornerDeck();
42          this.left1 = new SideDeck(Suit.CLUBS);
43          this.left2 = new SideDeck(Suit.DIAMONDS);
44          this.left3 = new SideDeck(Suit.HEARTS);
45          this.left4 = new SideDeck(Suit.SPADES);
46          this.right1 = new SideDeck(Suit.CLUBS);
47          this.right2 = new SideDeck(Suit.DIAMONDS);
48          this.right3 = new SideDeck(Suit.HEARTS);
49          this.right4 = new SideDeck(Suit.SPADES);
50          Deck[] decks = {upLeftCorner, downLeftCorner, upRightCorner,
51              downRightCorner, left1, left2, left3, left4, right1, right2, right3,
52              right4};
53          this.manoeuvreDecks = decks;
54      }
55  
56      public Deck[] getManoeuvreDecks() {
57          return manoeuvreDecks;
58      }    
59  
60      public final void createFoundationDecks() {
61          this.clubsDown = new FoundationDeck(Suit.CLUBS, Direction.DOWN);
62          this.clubsUp = new FoundationDeck(Suit.CLUBS, Direction.UP);
63          this.diamondsDown = new FoundationDeck(Suit.DIAMONDS, Direction.DOWN);
64          this.diamondsUp = new FoundationDeck(Suit.DIAMONDS, Direction.UP);
65          this.heartsDown = new FoundationDeck(Suit.HEARTS, Direction.DOWN);
66          this.heartsUp = new FoundationDeck(Suit.HEARTS, Direction.UP);
67          this.spadesDown = new FoundationDeck(Suit.SPADES, Direction.DOWN);
68          this.spadesUp = new FoundationDeck(Suit.SPADES, Direction.UP);
69  
70      }
71  
72      public final void addAcesAndKingsToFoundationDecks() {
73          clubsDown.addCard(new Card(Card.KING, Suit.CLUBS));
74          clubsUp.addCard(new Card(Card.ACE, Suit.CLUBS));
75          diamondsDown.addCard(new Card(Card.KING, Suit.DIAMONDS));
76          diamondsUp.addCard(new Card(Card.ACE, Suit.DIAMONDS));
77          heartsDown.addCard(new Card(Card.KING, Suit.HEARTS));
78          heartsUp.addCard(new Card(Card.ACE, Suit.HEARTS));
79          spadesDown.addCard(new Card(Card.KING, Suit.SPADES));
80          spadesUp.addCard(new Card(Card.ACE, Suit.SPADES));
81      }
82  
83      public final void dealToManoeuvreDecks() {
84          Dealer dealer = new Dealer();        
85          dealer.deal(this.manoeuvreDecks);
86  
87      }
