6    * The class creates and deals cards for manoeuvre decks.

9   public class Dealer {
10  
11      public Dealer() {
12      }
13  
14      public LinkedList<Card> createLesQuatreCoinsCards() {
15          // one set of aces and kings created in Game.createFoundationDecks()
16          LinkedList<Card> cards = createCards(1, 13);
17          cards.addAll(createCards(2, 12));
18          return cards;
19      }
20  
21      public void deal(Deck[] decks) {
22          LinkedList<Card> cards = createLesQuatreCoinsCards();
23          while (!cards.isEmpty()) {
24              for (Deck deck : decks) {
25                  if (cards.isEmpty()) {
26                      break;
27                  }
28                  deck.addCard(cards.pop());
29              }
30          }
31  
32      }
33  
34      public LinkedList createCards(int minValue, int maxValue) {
35          LinkedList<Card> cards = new LinkedList();
36          for (int i = minValue; i <= maxValue; i++) {
37              for (Suit suit : Suit.values()) {
38                  cards.add(new Card(i, suit));
39              }
40          }
41          return cards;
42      }
