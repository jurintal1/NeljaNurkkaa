10      public LinkedList<Card> createLesQuatreCoinsCards() {
11          // one set of aces and kings created in Game.createFoundationDecks()
12          LinkedList<Card> cards = createCards(1, 13);
13          cards.addAll(createCards(2, 12));
14          return cards;
15      }

17      public void deal(Deck[] decks) {
18          LinkedList<Card> cards = createLesQuatreCoinsCards();
19          while (!cards.isEmpty()) {
20              for (Deck deck : decks) {
21                  if (cards.isEmpty()) {
22                      break;
23                  }
24                  deck.addCard(cards.pop());
25              }
26          }
27  
28      }
29  
30      public LinkedList createCards(int minValue, int maxValue) {
31          LinkedList<Card> cards = new LinkedList();
32          for (int i = minValue; i <= maxValue; i++) {
33              for (Suit suit : Suit.values()) {
34                  cards.add(new Card(i, suit));
35              }
36          }
37          return cards;
38      }
