7   /**
8    * The class creates and deals cards for manoeuvre decks.
9    *
10   */
11  public class Dealer {
12  
13      /**
14       * A new dealer.
15       */
16      public Dealer() {
17      }
18  
19      /**
20       * Deals the LCC cards.
21       *
22       * @param decks where the cards are dealt
23       */
24      public void deal(Deck[] decks) {
25          LinkedList<Card> cards = createLesQuatreCoinsCards();
26          while (!cards.isEmpty()) {
27              for (Deck deck : decks) {
28                  if (cards.isEmpty()) {
29                      break;
30                  }
31                  deck.addCard(cards.pop());
32              }
33          }
34      }
35  
36      /**
37       * Creates and shuffles the cards for LCC. One set of aces and kings is
38       * created in Tableau.createFoundationDecks().
39       *
40       * @return the necessary cards as LinkedList.
41       */
42      public LinkedList<Card> createLesQuatreCoinsCards() {
43          // 
44          LinkedList<Card> cards = createCards(1, 13);
45          cards.addAll(createCards(2, 12));
46          long seed = System.nanoTime();
47          Collections.shuffle(cards, new Random(seed));
48          return cards;
49      }
50  
51      /**
52       * Creates Card objects between the defined values.
53       *
54       * @param minValue minimun face value in the cards created
55       * @param maxValue maximum face value in the cards created
56       * @return the cards as linked list
57       */
58      public LinkedList createCards(int minValue, int maxValue) {
59          LinkedList<Card> cards = new LinkedList();
60          for (int i = minValue; i <= maxValue; i++) {
61              for (Suit suit : Suit.values()) {
62                  cards.add(new Card(i, suit));
63              }
64          }
65          return cards;
66      }

68  }
