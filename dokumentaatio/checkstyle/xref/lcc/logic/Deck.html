3   import java.util.LinkedList;
4   
5   /**
6    * A class for initializing and accessing Deck objects. Deck is represented as
7    * LinkedList<Card>.
8    */
9   abstract public class Deck {
10  
11      protected LinkedList<Card> deck;
12  
13      public Deck() {
14          this.deck = new LinkedList<>();
15      }
16  
17      abstract boolean allowedToAdd(Card card);
18  
19      public void addCard(Card card) {
20          this.deck.push(card);
21      }
22  
23      public Card topCard() {
24          return this.deck.peek();
25      }
26  
27      public Card takeCard() {
28          return this.deck.pop();
29      }
30  
31      public int size() {
32          return this.deck.size();
33      }
34  
35      @Override
36      public String toString() {
37          return this.topCard().toString() + " - size " + this.size();
38      }
39  
40  }
