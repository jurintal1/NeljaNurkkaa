46      public void newGame(Stage window) {
47          this.window = window;
48          this.table = new Tableau();
49          this.gameplay = new Gameplay(table);

51          this.highScoreList = new HighScoreList();

54          this.mainView = createMainView();

61      /**
62       * Creates the game view.
63       *
64       * @return Scene
65       */
66      public Scene createMainView() {
67          BorderPane bPane = new BorderPane();
68          VBox rightBox = new VBox();
69          rightBox.setPadding(new Insets(100, 100, 100, 100));
70          rightBox.setSpacing(30);
71          rightBox.setBackground(new Background(new BackgroundFill(Color.FORESTGREEN, CornerRadii.EMPTY, Insets.EMPTY)));
72  
73          Label score = new Label("" + gameplay.getScore());
74          score.setTextFill(Color.WHITESMOKE);
75          score.setFont(new Font("Arial", 60));
76  
77          Button restart = new Button("Aloita alusta");
78          restart.setOnAction(e -> {
79              if (list.checkList(gameplay.getScore())) {
80                  window.setScene(mainView);
81              }
82              newGame(this.window);
83          });
84  
85          Button highscores = new Button("High Score -lista");
86          highscores.setOnAction(e -> {
87              try {
88                  window.setScene(createListView());
89              } catch (IOException ex) {
90                  System.out.println("ei onnistu");
91                  Logger.getLogger(GUI.class.getName()).log(Level.SEVERE, null, ex);
92              } catch (ClassNotFoundException ex) {
93                  Logger.getLogger(GUI.class.getName()).log(Level.SEVERE, null, ex);
94              }
95  
96          });
97  
98          rightBox.getChildren().addAll(score, restart, highscores);
99          bPane.setRight(rightBox);
100 
101         GridPane grid = new GridPane();
102         grid.setBackground(new Background(new BackgroundFill(Color.GREEN, CornerRadii.EMPTY, Insets.EMPTY)));
103         bPane.setLeft(grid);
104 
105         grid.setHgap(3);
106         grid.setVgap(3);
107         grid.setPadding(new Insets(10, 100, 10, 100));
108 
109         Tableau table = new Tableau();
110 
111         grid.add(new DeckView(table.getUpLeftCorner(), gameplay, score), 0, 0);
112         grid.add(new DeckView(table.getDownLeftCorner(), gameplay, score), 0, 3);
113 
114         grid.add(new DeckView(table.getLeft1(), gameplay, score), 1, 0);
115         grid.add(new DeckView(table.getLeft2(), gameplay, score), 1, 1);
116         grid.add(new DeckView(table.getLeft3(), gameplay, score), 1, 2);
117         grid.add(new DeckView(table.getLeft4(), gameplay, score), 1, 3);
118 
119         grid.add(new DeckView(table.getClubsUp(), gameplay, score), 2, 0);
120         grid.add(new DeckView(table.getDiamondsUp(), gameplay, score), 2, 1);
121         grid.add(new DeckView(table.getHeartsUp(), gameplay, score), 2, 2);
122         grid.add(new DeckView(table.getSpadesUp(), gameplay, score), 2, 3);
123 
124         grid.add(new DeckView(table.getClubsDown(), gameplay, score), 3, 0);
125         grid.add(new DeckView(table.getDiamondsDown(), gameplay, score), 3, 1);
126         grid.add(new DeckView(table.getHeartsDown(), gameplay, score), 3, 2);

179     /**
180      * Creates the view for high score list.
181      *
182      * @return the Scene with a view of high score list
183      * @throws IOException file can't be read
184      * @throws FileNotFoundException file not found
185      * @throws ClassNotFoundException can't read ArrayList from File
186      */
187     public Scene createListView() throws IOException, FileNotFoundException, ClassNotFoundException {
188 
189         TextArea text = new TextArea();
190         ArrayList<HighScore> scores = highScoreList.getListFromFile();
191         for (int i = 0; i < 10; i++) {
192 
193             text.appendText(scores.get(i).toString() + "\n");
194         }
195         BorderPane bp = new BorderPane(text);
196         return new Scene(bp);
197     }
